// ============================================================ // 현금 마감 자동화 - Code.gs (GitHub Pages 연동 버전) // ============================================================ function getConfig() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('설정'); const data = sheet.getDataRange().getValues(); const config = {}; data.forEach(row => { if (row[0]) config[row[0]] = row[1]; }); return config; } // ============================================================ // 🌐 GitHub Pages → 앱스크립트 API 진입점 // ============================================================ function doGet(e) { const action = e.parameter.action; let result; if (action === 'getPrev') { result = getPrevCoinData(); } else { result = { status: 'ok' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const action = e.parameter.action; let result; if (action === 'save') { const formData = JSON.parse(e.postData.contents); result = saveSettlement(formData); } else { result = { error: 'unknown action' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ============================================================ // 📋 전일 동전 데이터 불러오기 // ============================================================ function getPrevCoinData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('정산현황'); const data = sheet.getDataRange().getValues(); if (data.length < 2) { return { c10:0, c50:0, c100:0, c500:0, prevTotal:0, prevDate:'데이터 없음' }; } const headers = data[0]; const last = data[data.length - 1]; const idx = (n) => headers.indexOf(n); const prevDate = last[idx('날짜')]; const prevDateStr = prevDate instanceof Date ? (prevDate.getMonth()+1) + '월 ' + prevDate.getDate() + '일' : String(prevDate).split(' ')[0]; return { c10: last[idx('10원')] || 0, c50: last[idx('50원')] || 0, c100: last[idx('100원')] || 0, c500: last[idx('500원')] || 0, prevTotal: last[idx('실제잔액')] || 0, prevDate: prevDateStr }; } // ============================================================ // 💾 마감 데이터 저장 // ============================================================ function saveSettlement(formData) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('정산현황'); const feedback = ss.getSheetByName('피드백'); const now = new Date(); const realTotal = formData.c10 * 10 + formData.c50 * 50 + formData.c100 * 100 + formData.c500 * 500 + formData.b1000 * 1000 + formData.b5000 * 5000 + formData.b10000 * 10000 + formData.b50000 * 50000 + formData.b100000 * 100000 + formData.b1000000 * 1000000 + formData.plusAmount - formData.minusAmount + (formData.missAmount || 0); const theoryTotal = formData.prevTotal + formData.plusAmount - formData.minusAmount + (formData.missAmount || 0); const diff = realTotal - theoryTotal; sheet.appendRow([ now, formData.manager, formData.c10, formData.c50, formData.c100, formData.c500, formData.b1000, formData.b5000, formData.b10000, formData.b50000, formData.b100000, formData.b1000000, formData.plusAmount, formData.plusReason, formData.minusAmount, formData.minusReason, formData.missAmount || 0, formData.missReason || '', realTotal, theoryTotal, diff, formData.needCoin ? 'Y' : '', formData.needBill ? 'Y' : '', formData.memo ]); if (formData.fbEfficiency || formData.fbCustomer || formData.fbMaintenance) { feedback.appendRow([ now, formData.manager, formData.fbEfficiency || '', formData.fbCustomer || '', formData.fbMaintenance || '' ]); } sendReconciliationSms(formData, realTotal, theoryTotal, diff); if (formData.b50000 >= 10) sendDepositReminder(formData.b50000); return { success: true, diff: diff, realTotal: realTotal }; } // ============================================================ // 📱 SENS 문자 발송 // ============================================================ function sendSms(to, content) { const config = getConfig(); const serviceId = config['SENS_SERVICE_ID']; const accessKey = config['SENS_ACCESS_KEY']; const secretKey = config['SENS_SECRET_KEY']; const sender = config['SENS_SENDER']; const url = 'https://sens.apigw.ntruss.com/sms/v2/services/' + serviceId + '/messages'; const timestamp = String(new Date().getTime()); const message = 'POST /sms/v2/services/' + serviceId + '/messages\n' + timestamp + '\n' + accessKey; const signature = Utilities.base64Encode( Utilities.computeHmacSha256Signature(message, secretKey) ); const payload = JSON.stringify({ type: 'SMS', from: sender, content: content, messages: [{ to: String(to) }] }); try { UrlFetchApp.fetch(url, { method: 'post', contentType: 'application/json', headers: { 'x-ncp-apigw-timestamp': timestamp, 'x-ncp-iam-access-key': accessKey, 'x-ncp-apigw-signature-v2': signature }, payload: payload, muteHttpExceptions: true }); } catch(e) { Logger.log('SMS 오류: ' + e.message); } } function sendSmsToManagers(content) { const config = getConfig(); sendSms(config['PHONE_SIWON'], content); sendSms(config['PHONE_BADA'], content); } function sendSmsToOperators(content) { const config = getConfig(); sendSms(config['PHONE_BADA'], content); sendSms(config['PHONE_DAHYUN'], content); } function sendReconciliationSms(formData, realTotal, theoryTotal, diff) { const today = Utilities.formatDate(new Date(), 'Asia/Seoul', 'M/d(EEE)'); const diffText = diff === 0 ? '✅ 일치' : (diff > 0 ? '⚠️ +' : '⚠️ ') + Math.abs(diff).toLocaleString() + '원 ' + (diff > 0 ? '초과' : '부족'); const msg = '[현금마감] ' + today + ' ' + formData.manager + '\n' + '─────────────\n' + '실제잔액: ' + realTotal.toLocaleString() + '원\n' + '이론잔액: ' + theoryTotal.toLocaleString() + '원\n' + '차이: ' + diffText + '\n' + (formData.plusAmount ? '증가: +' + formData.plusAmount.toLocaleString() + '원 (' + formData.plusReason + ')\n' : '') + (formData.minusAmount ? '감소: -' + formData.minusAmount.toLocaleString() + '원 (' + formData.minusReason + ')\n' : '') + (formData.missAmount ? '누락: ' + formData.missAmount.toLocaleString() + '원 (' + formData.missReason + ')\n' : '') + (formData.needCoin ? '🪙 동전 보충 필요\n' : '') + (formData.needBill ? '💵 지폐 보충 필요\n' : '') + (formData.memo ? '메모: ' + formData.memo : ''); sendSmsToManagers(msg); } function sendDepositReminder(count) { const config = getConfig(); sendSms(config['PHONE_BADA'], '[입금알림] 5만원권 ' + count + '장 보유 중\n입금 처리가 필요합니다.'); } // ============================================================ // ⏰ 리마인드 트리거 함수들 // ============================================================ function remind2300() { if (new Date().getDay() === 1) return; sendSmsToOperators('[긴급] ' + d + ' 현금 마감 누락\n전날 현금 마감이 완료되지 않았습니다.'); } } function remind1500() { const today = new Date(); const dayOfWeek = today.getDay(); if (dayOfWeek === 1) return; // 월요일 제외 const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('정산현황'); const data = sheet.getDataRange().getValues(); if (data.length < 2) return; const headers = data[0]; // 화요일이면 전전날(일요일) 마감 기준, 나머지는 마지막 행 기준 let targetRow; if (dayOfWeek === 2) { // 일요일 마감 행 찾기 const sunday = new Date(today); break; } } } } else { targetRow = data[data.length - 1]; } if (!targetRow) return; const needCoin = targetRow[headers.indexOf('동전보충필요')] === 'Y'; const needBill = targetRow[headers.indexOf('지폐보충필요')] === 'Y'; if (needBill) items.push('💵 지폐'); sendSmsToManagers('[잔돈 보충 알림] ' + d + ' 마감 기준\n' + items.join(', ') + ' 보충이 필요합니다.'); } function isTodaySettled() { return isDateSettled(new Date()); } function isYesterdaySettled() { const y = new Date(); const target = Utilities.formatDate(targetDate, 'Asia/Seoul', 'yyyy/MM/dd'); for (let i = 1; i < data.length; i++) { if (data[i][0] instanceof Date) { if (Utilities.formatDate(data[i][0], 'Asia/Seoul', 'yyyy/MM/dd') === target) return true; } } return false; } // ============================================================ // 🔧 최초 1회 실행 함수들 // ============================================================ function createAllTriggers() { ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t));
